pipeline {
    agent any

    environment {
        DOCKER_IMAGE = "property-view"
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        stage('Build') {
            steps {
                echo 'Сборка проекта...'
                sh 'mvn clean compile'
            }
        }
        stage('Test') {
            steps {
                echo 'Запуск тестов...'
                sh 'mvn test'
            }
            post {
                always {
                    junit '**/target/surefire-reports/*.xml'
                }
            }
        }
        stage('Package') {
            steps {
                echo 'Упаковка проекта...'
                sh 'mvn package -DskipTests'
            }
        }
    }

    post {
        always {
            echo 'Pipeline завершён.'
        }
    }
}
